//Copy Equipment Category Configuration in Equipment Configuration

CREATE OR REPLACE TRIGGER EQP_CAT_CONFIG
AFTER INSERT
ON MAINTENANCE.MACHINE_CATS_CONFIG
REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE 
 i NUMBER;
 counter NUMBER;
 eqpID maintainable_unit.ID%TYPE;
 
 CURSOR equipments IS SELECT ID FROM "MAINTENANCE"."MAINTAINABLE_UNIT" WHERE PARENT_ID = :NEW.EQUIPMENT_CAT_ID;
BEGIN
 SELECT COUNT(ID) INTO counter FROM MAINTAINABLE_UNIT WHERE PARENT_ID = :NEW.EQUIPMENT_CAT_ID;
 
 DBMS_OUTPUT.PUT_LINE('Counter = '|| counter);
 
 OPEN equipments;

 FOR i IN 1 .. counter LOOP
  FETCH equipments INTO eqpID;
  DBMS_OUTPUT.PUT_LINE(:NEW.ID+i ||' ' ||eqpID||' '||:NEW.ITEM_ID||' '||:NEW.CATEGORY_NAME||' '||:NEW.ITEM_DSCRPTN||' '||:NEW.ITEM_PRICE);
  INSERT INTO MCHN_ITEM_CATEGORIES VALUES (:NEW.ID+i, eqpID , :NEW.ITEM_ID, :NEW.CATEGORY_NAME, :NEW.ITEM_DSCRPTN, :NEW.ITEM_PRICE);
 END LOOP;
 
 CLOSE equipments;
 
 DBMS_OUTPUT.PUT_LINE('Right Insertion');
END;
